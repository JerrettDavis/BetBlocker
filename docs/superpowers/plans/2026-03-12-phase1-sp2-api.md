```rust
use sqlx::PgPool;
use jsonwebtoken::{DecodingKey, EncodingKey};
use std::sync::Arc;

#[derive(Clone)]
pub struct AppState {
    pub db: PgPool,
    pub redis: redis::Client,
    pub jwt_encoding_key: Arc<EncodingKey>,
    pub jwt_decoding_key: Arc<DecodingKey>,
    pub config: Arc<ApiConfig>,
}

impl AppState {
    pub async fn new(config: ApiConfig) -> anyhow::Result<Self> {
        let db = PgPool::connect(&config.database_url).await?;
        let redis = redis::Client::open(config.redis_url.as_str())?;

        // Load Ed25519 keys from PEM files
        let private_key_pem = std::fs::read(&config.jwt_private_key_path)?;
        let public_key_pem = std::fs::read(&config.jwt_public_key_path)?;
        let jwt_encoding_key = Arc::new(EncodingKey::from_ed_pem(&private_key_pem)?);
        let jwt_decoding_key = Arc::new(DecodingKey::from_ed_pem(&public_key_pem)?);

        Ok(Self {
            db,
            redis,
            jwt_encoding_key,
            jwt_decoding_key,
            config: Arc::new(config),
        })
    }
}
```

## Chunk 2: Auth Module

### Task 5: Auth Service (Password + JWT + Refresh Tokens)

**Files:** `crates/bb-api/src/services/auth_service.rs`

- [ ] **Step 1: Password hashing.** Use `bcrypt` crate with cost factor 12. Implement `hash_password(plain: &str) -> Result<String>` and `verify_password(plain: &str, hash: &str) -> Result<bool>`.

- [ ] **Step 2: Password validation.** Enforce minimum 12 chars, must contain uppercase, lowercase, digit, and special character. Return structured validation errors listing which rules failed.

- [ ] **Step 3: JWT token issuance.** Define `Claims { sub: Uuid, email: String, role: String, iat: i64, exp: i64, jti: Uuid }`. Implement `issue_access_token(account: &Account, state: &AppState) -> Result<(String, i64)>` that creates a JWT signed with Ed25519 via `jsonwebtoken::encode` using `Algorithm::EdDSA`. Return the token string and `expires_in` seconds.

- [ ] **Step 4: Refresh token management.** Generate 256-bit random refresh token (hex-encoded, prefixed `rtk_`). Store SHA-256 hash in `refresh_tokens` table with `account_id`, `token_family` (UUID), `user_agent`, `ip_prefix`, `expires_at`, `used` (bool). On refresh: look up hash, verify not expired and not used, mark as used, issue new refresh token in same family. If a used token is presented (reuse detection), revoke the entire family and return `TOKEN_FAMILY_REVOKED`.

- [ ] **Step 5: Account lockout.** Track failed login attempts in Redis keyed by email. After 5 consecutive failures, set a 15-minute lockout. Clear on successful login.

### Task 6: Auth Route Handlers

**Files:** `crates/bb-api/src/routes/auth.rs`

- [ ] **Step 2: POST /auth/login.** Look up account by email. Check lockout status. Verify password. If failed, increment failure counter and return generic `INVALID_CREDENTIALS`. On success, clear failure counter, issue tokens, return 200.

- [ ] **Step 5: POST /auth/forgot-password.** Look up account by email. If found, generate 256-bit reset token, store hash with 1-hour expiry. Always return 202 (prevent enumeration). Log the token for Phase 1.

- [ ] **Step 6: POST /auth/reset-password.** Validate reset token hash. Verify not expired. Hash new password, update account, delete reset token, revoke all refresh tokens for the account. Return 200.

### Task 7: JWT Auth Middleware and Extractor

**Files:** `crates/bb-api/src/middleware/auth.rs`, `crates/bb-api/src/extractors.rs`

- [ ] **Step 1: Define AuthenticatedAccount and implement FromRequestParts.**

```rust
use axum::extract::FromRequestParts;
use axum::http::request::Parts;
use jsonwebtoken::{decode, Algorithm, Validation};
use uuid::Uuid;

#[derive(Debug, Clone)]
pub struct AuthenticatedAccount {
    pub account_id: Uuid,
    pub email: String,
    pub role: String,
}

impl<S> FromRequestParts<S> for AuthenticatedAccount
where
    S: Send + Sync,
    AppState: FromRef<S>,
{
    type Rejection = ApiError;

    async fn from_request_parts(parts: &mut Parts, state: &S) -> Result<Self, Self::Rejection> {
        let app_state = AppState::from_ref(state);

        let auth_header = parts
            .headers
            .get("Authorization")
            .and_then(|v| v.to_str().ok())
            .and_then(|v| v.strip_prefix("Bearer "))
            .ok_or(ApiError::Unauthorized {
                code: "UNAUTHORIZED".into(),
                message: "Missing or malformed Authorization header".into(),
            })?;

        let mut validation = Validation::new(Algorithm::EdDSA);
        validation.set_issuer(&["betblocker-api"]);

        let token_data = decode::<Claims>(
            auth_header,
            &app_state.jwt_decoding_key,
            &validation,
        )
        .map_err(|e| ApiError::Unauthorized {
            code: "INVALID_TOKEN".into(),
            message: format!("Invalid or expired token: {e}"),
        })?;

        Ok(AuthenticatedAccount {
            account_id: token_data.claims.sub,
            email: token_data.claims.email,
            role: token_data.claims.role,
        })
    }
}
```

- [ ] **Step 5: POST /enrollments/:id/unenroll.** Implement enrollment policy enforcement:

```rust
pub async fn request_unenrollment(
    db: &PgPool,
    enrollment: &Enrollment,
    requester_id: Uuid,
    reason: Option<String>,
) -> Result<UnenrollmentResult, ApiError> {
    // Only the enrollment owner (account_id) can request unenrollment
    if enrollment.account_id != requester_id {
        return Err(ApiError::Forbidden {
            message: "Only the enrolled account can request unenrollment".into(),
        });
    }

    if enrollment.status != EnrollmentStatus::Active {
        return Err(ApiError::Conflict {
            code: "ENROLLMENT_NOT_ACTIVE".into(),
            message: "Enrollment is not active".into(),
        });
    }

    match &enrollment.unenrollment_policy.policy_type {
        UnenrollmentType::TimeDelayed => {
            let cooldown_hours = enrollment.unenrollment_policy.cooldown_hours
                .unwrap_or(48);
            let eligible_at = Utc::now() + Duration::hours(cooldown_hours as i64);

            // Insert unenrollment request with eligible_at set
            insert_unenroll_request(db, enrollment.id, requester_id, reason, Some(eligible_at)).await?;
            update_enrollment_status(db, enrollment.id, EnrollmentStatus::UnenrollRequested).await?;

            Ok(UnenrollmentResult::TimeDelayed { eligible_at })
        }
        UnenrollmentType::PartnerApproval => {
            let approver_id = enrollment.unenrollment_policy.requires_approval_from
                .ok_or(ApiError::Internal {
                    message: "Partner enrollment missing approval authority".into(),
                })?;

            // Insert unenrollment request, eligible_at is None (needs approval)
            insert_unenroll_request(db, enrollment.id, requester_id, reason, None).await?;
            update_enrollment_status(db, enrollment.id, EnrollmentStatus::UnenrollRequested).await?;

            // TODO: notify partner (email/push) -- for Phase 1, log it
            Ok(UnenrollmentResult::PendingApproval { approver_id })
        }
        UnenrollmentType::AuthorityApproval => {
            let approver_id = enrollment.unenrollment_policy.requires_approval_from
                .ok_or(ApiError::Internal {
                    message: "Authority enrollment missing approval authority".into(),
                })?;

            insert_unenroll_request(db, enrollment.id, requester_id, reason, None).await?;
            update_enrollment_status(db, enrollment.id, EnrollmentStatus::UnenrollRequested).await?;

            Ok(UnenrollmentResult::PendingApproval { approver_id })
        }
    }
}
```
